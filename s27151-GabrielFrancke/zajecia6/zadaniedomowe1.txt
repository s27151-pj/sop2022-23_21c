#include <stdlib.h>
#include <stdio.h>
#include <signal.h>

void signalHandler(int sig) {
    if (sig == SIGUSR1) {
        printf("Otrzymano sygna≈Ç: %d\n", sig);
        exit(1);
    }
}

int main() {
    int pid = getpid();

    printf("PID to %d\n", pid);

    while (1) {
        signal(SIGUSR1, signalHandler);
    }

    return 0;
}
